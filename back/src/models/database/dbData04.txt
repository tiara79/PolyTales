CREATE TABLE tutor (
  chatid SERIAL PRIMARY KEY,  -- 자동 증가 채팅 ID
  userid INTEGER NOT NULL REFERENCES users(userid),  -- 사용자 ID (NOT NULL 추천)
  storyid INTEGER NOT NULL REFERENCES story(storyid),  -- 스토리 참조 (NOT NULL 추천)
  sender TEXT CHECK (sender IN ('user', 'tutor')) NOT NULL,  -- 발신자 구분
  message TEXT NOT NULL,  -- 메시지 내용
  createdat TIMESTAMP DEFAULT CURRENT_TIMESTAMP  -- 작성 시각
);
