
# 테이블에 status, role, plan,pay 컬럼 추가

ALTER TABLE users
ADD COLUMN status INTEGER CHECK (status IN (1, 2, 3)) DEFAULT 1,  -- 1:정상, 2:경고, 3:강제삭제
ADD COLUMN role INTEGER CHECK (role IN (1, 2)) DEFAULT 2,        -- 1:관리자, 2:회원
ADD COLUMN plan INTEGER CHECK (plan IN (1, 2, 3)) DEFAULT 1,     -- 1:무료, 2:스탠다드, 3:프리미엄
ADD COLUMN pay INTEGER CHECK (pay IN (1, 2, 3));
